Longest Word

This program is designed to search through a text file and display the longest word within it.

'use strict'; angular.module('rg_personal').controller('projects-longestCtrl', function () { var self = this; this.content = ''; this.result = ''; this.getTxt = function ($fileContent) { self.content = $fileContent; } this.calculate = function () { var arr = self.content.split(new RegExp(/[\s,()\n\r\t]+/)); var longest = ''; for (var i = 0; i < arr.length; i++) { var current = arr[i]; if (current.length > longest.length) longest = current; } self.result = "The longest word in the txt file is: " + longest; } })
