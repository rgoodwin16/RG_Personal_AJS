FizzBuzz

This program is designed to display all the numbers between 1 & 100, with every number that is a multiple of 3 being replaced with the word "Fizz", every number that is a multiple of 5 being replaced with the word "Buzz", and finally any number that is a multiple of both 3 & 5 being replaced with the word "FizzBuzz".

            
                //FizzBuzz
                function fizzBuzz() {
                    var fizzHtml = document.getElementById('resultFizz');
                    for (i = 1; i <= 100; i++) {
                        if ((i % 3 == 0) && (i % 5 == 0)) {
                            fizzHtml.innerHTML += "FizzBuzz" + " ";
                        } else if ((i % 3 == 0)) {
                            fizzHtml.innerHTML += "Fizz" + " ";
                        } else if ((i % 5 == 0)) {
                            fizzHtml.innerHTML += "Buzz" + " ";
                        } else {
                            fizzHtml.innerHTML += i + " ";
                        }
                    }
                }
